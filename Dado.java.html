public class Dado {
    private int quantLados;
    private int valorFace;

    public Dado(int quantLados) {
        valorFace = 1;
        this.quantLados = quantLados;

    }

    public int getValorFace()
    {
        return valorFace;
    }

    public void setValorFace(int valorFace){
        if (valorFace <= 6 && valorFace > 0) {
            this.valorFace = valorFace;
        } else {
            System.out.println("Valor "+ valorFace + " inv√°lido para a face do dado que tem " + quantLados + " lados.\n");
        }

    }

    public void rolarDado(){
        int ladoRolado = (int) (Math.random() * quantLados) + 1;
        System.out.println("Valor rolado: " + ladoRolado);
        this.valorFace = ladoRolado;

    }

    public int getQuantidadeLados(){
        return this.quantLados;
    }


}
